@Lazy

Tout bean dans "shared" devrait être @Lazy

@Lazy @Autowire


@Service
class Foo {
	private final Bar bar;

	@Autowire
	public Foo(@Lazy Bar bar) {
		this.bar = bar; // peut être null
	}
}

Tests


class MyTest {
    @Test
    public void test() {
        // GIVEN
        ConfigurationManager.getConfigInstance()
            .setProperty("foo", "bar");

        // WHEN
        String myVar = DynamicPropertyFactory.getInstance()
            .getStringProperty("foo", null)
            .get();

        // THEN
        assertThat(myVar).isEqualsTo("bar");
    }

    @After
    public void after() {
        ConfigurationManager.getConfigInstance().clear();
    }
}
